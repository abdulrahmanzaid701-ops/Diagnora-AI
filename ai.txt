<script>
(function(){
  const btn = document.getElementById('runBtn');
  const input = document.getElementById('symptomInput');
  const traceLog = document.getElementById('traceLog');
  const traceStatus = document.getElementById('traceStatus');
  const answerBody = document.getElementById('answerBody');

  const traceSteps = [
    { tag: '[intake]',   text: 'Parsing description into structured signs…' },
    { tag: '[search]',   text: 'Searching clinical references for matching patterns…' },
    { tag: '[search]',   text: 'Pulling related literature from PubMed and NHS guidance…' },
    { tag: '[cross-chk]',text: 'Weighing candidates against onset, location, and severity…' },
    { tag: '[cross-chk]',text: 'Ruling out poor fits, ranking what remains…' },
    { tag: '[report]',   text: 'Assembling findings and attaching sources…' },
  ];

  const exampleFindings = [
    {
      name: 'Appendicitis',
      fit: 'strong fit',
      desc: 'Pain that starts near the navel and migrates to the lower right abdomen, worsened by pressure, is a classic pattern — especially with fever.',
      cite: 'source: NHS — Appendicitis, Mayo Clinic Symptoms & Causes'
    },
    {
      name: 'Ovarian cyst (if applicable)',
      fit: 'possible fit',
      desc: 'Can cause similar one-sided lower abdominal pain; usually distinguished by menstrual timing and pain quality.',
      cite: 'source: NIH — Ovarian Cysts Overview'
    },
    {
      name: 'Gastroenteritis',
      fit: 'weaker fit',
      desc: 'Common cause of abdominal discomfort, but typically comes with more diffuse pain and digestive symptoms rather than a fixed, tender point.',
      cite: 'source: Cochrane Library — Acute Gastroenteritis Review'
    }
  ];

  let running = false;

  function resetPanes(){
    traceLog.innerHTML = '';
    answerBody.innerHTML = '<p class="answer-empty">Running research…</p>';
  }

  function setStatus(text, active){
    traceStatus.innerHTML = '<span class="pulse" style="' + (active ? '' : 'animation:none;opacity:.4;') + '"></span>' + text;
  }

  function appendLine(step){
    return new Promise(resolve => {
      const line = document.createElement('div');
      line.className = 'trace-line';
      line.innerHTML = '<span class="tag">' + step.tag + '</span>' + step.text;
      traceLog.appendChild(line);
      setTimeout(() => {
        line.classList.add('done');
        resolve();
      }, 520);
    });
  }

  function renderAnswer(){
    let html = '';
    exampleFindings.forEach(f => {
      html += '<div class="finding">' +
        '<div class="finding-top"><span class="finding-name">' + f.name + '</span><span class="finding-fit mono">' + f.fit + '</span></div>' +
        '<div class="finding-desc">' + f.desc + '</div>' +
        '<div class="cite">' + f.cite + '</div>' +
      '</div>';
    });
    html += '<div class="flag-box"><strong>Seek urgent care if</strong>Pain becomes severe or constant, you develop a high fever, or you can\'t keep fluids down. These can signal something that needs same-day attention.</div>';
    answerBody.innerHTML = html;
  }

  async function runResearch(){
    if(running) return;
    running = true;
    btn.disabled = true;
    btn.textContent = 'Researching…';
    resetPanes();
    setStatus('running', true);

    for(let i = 0; i < traceSteps.length; i++){
      await appendLine(traceSteps[i]);
      traceLog.scrollTop = traceLog.scrollHeight;
    }

    setStatus('complete', false);
    renderAnswer();
    btn.disabled = false;
    btn.textContent = 'Research this';
    running = false;
  }

  btn.addEventListener('click', runResearch);
  input.addEventListener('keydown', e => {
    if(e.key === 'Enter' && (e.metaKey || e.ctrlKey)) runResearch();
  });
})();
</script>
